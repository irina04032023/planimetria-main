            function roundValue(value) {

            function areaSquare(a) {
                if (a <= 0) return null;
                return a * a;
            }

            function areaSquareByDiagonal(d) {
                if (d <= 0) return null;
                return (d * d) / 2;
            }

            function areaTriangleBaseHeight(a, h) {
                if (a <= 0 || h <= 0) return null;
                return (a * h) / 2;
            }

            function areaTriangleTwoSidesAngle(a, b, sinGamma) {
                if (a <= 0 || b <= 0 || sinGamma <= 0) return null;
                return 0.5 * a * b * sinGamma;
            }

            function areaTriangleHeron(a, b, c) {
                if (a <= 0 || b <= 0 || c <= 0) return null;
                if (a + b <= c || a + c <= b || b + c <= a) return null;
                let p = (a + b + c) / 2;
                return Math.sqrt(p * (p - a) * (p - b) * (p - c));
            }

            function areaParallelogramBaseHeight(a, h) {
                if (a <= 0 || h <= 0) return null;
                return a * h;
            }

            function areaParallelogramSidesAngle(a, b, sinAlpha) {
                if (a <= 0 || b <= 0 || sinAlpha <= 0) return null;
                return a * b * sinAlpha;
            }

            function areaTrapezoidBasesHeight(a, b, h) {
                if (a <= 0 || b <= 0 || h <= 0) return null;
                return ((a + b) / 2) * h;
            }

            function areaTrapezoidMidlineHeight(m, h) {
                if (m <= 0 || h <= 0) return null;
                return m * h;
            }

            function areaRhombusSideHeight(a, h) {
                if (a <= 0 || h <= 0) return null;
                return a * h;
            }

            function areaRhombusSideAngle(a, sinAlpha) {
                if (a <= 0 || sinAlpha <= 0) return null;
                return a * a * sinAlpha;
            }

            function areaRhombusDiagonals(d1, d2) {
                if (d1 <= 0 || d2 <= 0) return null;
                return (d1 * d2) / 2;
            }

            function areaCircleRadius(r) {
                if (r <= 0) return null;
                return Math.PI * r * r;
            }

            function areaCirclePerimeter(p) {
                if (p <= 0) return null;
                return (p * p) / (4 * Math.PI);
            }

                if (a <= 0) return null;
                return 4 * a;
            }

            function perimeterRhombus(a) {
                if (a <= 0) return null;
                return 4 * a;
            }

            function perimeterTriangle(a, b, c) {
                if (a <= 0 || b <= 0 || c <= 0) return null;
                if (a + b <= c || a + c <= b || b + c <= a) return null;
                return a + b + c;
            }

            function perimeterParallelogram(a, b) {
                if (a <= 0 || b <= 0) return null;
                return 2 * (a + b);
            }

            function perimeterTrapezoid(a, b, c, d) {
                if (a <= 0 || b <= 0 || c <= 0 || d <= 0) return null;
                return a + b + c + d;
            }

            function perimeterCircle(r) {
                if (r <= 0) return null;
                return 2 * Math.PI * r;
            }
